#include<iostream>
#include<string>
#include<cstring>
using namespace std;
int main() {
	int a[501], b[501], len;
	memset(a, 0, sizeof(a));
	memset(b, 0, sizeof(b));
	string one, two;
	cin >> one >> two;
	a[0] = one.length();
	b[0] = two.length();
	len = (a[0] > b[0] ? a[0] : b[0]) + 1;
	for (int i = 1; i <= a[0]; i++) {
		a[i] = one[a[0] - i] - '0';
	}
	for (int i = 1; i <= b[0]; i++) {
		b[i] = two[b[0] - i] - '0';
	}
	for (int i = 1; i <= len; i++) {
		a[i] += b[i];
		a[i + 1] += a[i] / 10;
		a[i] %= 10;
	}
	while (a[len] == 0 && len > 1) {
		len--;
	}
	a[0] = len;
	for (int i = len; i > 0; i--) {
		cout << a[i];
	}
	return 0;
}
